In case you don't want to use the connected account to execute a tool, you can
provide the required auth params manually.

<CodeGroup>
    ```python Python
    from composio import Composio

    # Initialize the Composio client
    composio = Composio(api_key="<COMPOSIO_API_KEY>")

    # Execute the tool with custom auth params
    response = composio.tools.execute(
        user_id="default",
        slug="GITHUB_STAR_A_REPOSITORY_FOR_THE_AUTHENTICATED_USER",
        arguments={
            "owner": "composiohq",
            "repo": "composio",
        },
        custom_auth_params={
            "parameters": [
                {
                    "name": "Authorization",
                    "value": "Bearer <ACCESS_TOKEN>",
                    "in": "header",
                },
                {
                    "name": "Accept",
                    "value": "application/vnd.github.v3+json",
                    "in": "header",
                },
            ],
        },
    )
    print(response)
    ```
    ```typescript TypeScript
    import { Composio } from "@composio/sdk";

    // Initialize the Composio client
    const composio = new Composio({apiKey: "<COMPOSIO_API_KEY>"});

    // Execute the tool with custom auth params
    const response = await composio.tools.execute("GITHUB_STAR_A_REPOSITORY_FOR_THE_AUTHENTICATED_USER", {
        arguments: {
            owner: "composiohq",
            repo: "composio",
        },
        customAuthParams: {
            parameters: [
                {
                    name: "Authorization",
                    value: "Bearer <ACCESS_TOKEN>",
                    in: "header",
                },
                {
                    name: "Accept",
                    value: "application/vnd.github.v3+json",
                    in: "header",
                },
            ],
        },
    });
    ```
</CodeGroup>

## Using modifiers

You can also use modifiers to add custom auth params at the runtime in case you
don't want to use the connected account to execute a tool.

<CodeGroup>
    ```python Python
    from composio import Composio, before_execute
    from composio.types import ToolExecuteParams

    # Initialize the Composio client
    composio = Composio(api_key="<COMPOSIO_API_KEY>")

    @before_execute(toolkits=["NOTION"])
    def add_custom_auth(
        tool: str,
        toolkit: str,
        params: ToolExecuteParams,
    ) -> ToolExecuteParams:
        params["custom_auth_params"] = {
            "parameters": [
                {
                    "name": "x-api-key",
                    "value": os.getenv("NOTION_API_KEY"),
                    "in": "header",
                },
            ],
        }
        return params

    tools = composio.tools.get(
        user_id="default",
        toolkits=["NOTION"],
        modifiers=[add_custom_auth],
    )
    ```
    ```typescript TypeScript
    import { Composio } from "@composio/sdk";

    // Initialize the Composio client
    const composio = new Composio({apiKey: "<COMPOSIO_API_KEY>"});

    const tools = await composio.tools.get("default", {
        toolkits: ["NOTION"],
        modifiers: {
            beforeExecute:  ({ toolSlug, toolkitSlug, params }) => {
                if (toolkitSlug !== "NOTION") {
                    return params;
                }

                params.customAuthParams = {
                    parameters: [
                        {
                            name: "x-api-key",
                            value: process.env.NOTION_API_KEY,
                            in: "header",
                        },
                    ],
                };
                return params;
            },
        },
    });
    ```
</CodeGroup>
